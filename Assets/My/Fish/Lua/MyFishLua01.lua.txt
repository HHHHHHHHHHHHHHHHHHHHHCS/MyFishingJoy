
local GameObject = CS.UnityEngine.GameObject;
local Vector3 = CS.UnityEngine.Vector3;

--1.1 点击宝箱的金币钻石太拥挤分散一点
xlua.hotfix(CS.Treasour, 'CreatePrize',
    function (self)
    	print("HotFix ： CreatePrize");
        for i = 1, 5, 1 do
            local go = GameObject.Instantiate(self.gold, self.transform.position + Vector3(-10 + i * 30, 0, 0), self.transform.rotation);
            go.transform:SetParent(self.cavas);
            local go1 = GameObject.Instantiate(self.diamands, self.transform.position + Vector3(0, 30, 0) + Vector3(-10 + i * 30, 0, 0), self.transform.rotation);
            go1.transform:SetParent(self.cavas);
        end
    end);
